1. Tracking Files Lifecycle
git init
touch app.txt
git status
git add app.txt
git commit -m "Initial commit"

# modify file
echo "update" >> app.txt
git diff

git add app.txt
git commit -m "Updated file"

2. Branching and Isolation
git branch dev
git checkout dev

# changes in dev
echo "dev work" >> app.txt

git checkout main
git diff main dev

3. Merging and Conflict Handling
git checkout main
git merge dev

# conflict resolve (manual edit)

git add .
git commit -m "resolved conflict"


4. Working with Remote Repositories
git remote add origin <url>

git clone <url>

git fetch origin
git pull origin main

git push origin main

5. GitHub Collaboration Workflow
git init
git add .
git commit -m "first commit"

git remote add origin <url>
git push -u origin main

git checkout -b feature
git push origin feature

6. Undoing Changes & Recovery
git checkout -- app.txt

git reset app.txt

git commit --amend

git reset --soft HEAD~1
git reset --mixed HEAD~1
git reset --hard HEAD~1

git revert HEAD

git reflog

7. Rollback & Recovery Operations
git log

git reset --hard <commit-id>

git revert <commit-id>

git status
